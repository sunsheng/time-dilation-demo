function animate(timestamp) { if (!lastTimestamp) lastTimestamp = timestamp; const deltaMs = timestamp - lastTimestamp; lastTimestamp = timestamp; // 将动画时间缩放到一个合理的模拟时间步长（deltaSeconds） const deltaSeconds = deltaMs / 1000; // 1:1 实时，也可以乘个倍率加速 // 更新实验计时：地面时间以 deltaSeconds 前进，飞船以 deltaSeconds / γ 前进 if (experimentRunning) { const gamma = 1 / Math.sqrt(1 - vOverC * vOverC || 1e-8); labTime += deltaSeconds; shipTime += deltaSeconds / gamma; updateExperimentTexts(); } drawRestClock(timestamp); drawMovingClock(timestamp); requestAnimationFrame(animate); }
